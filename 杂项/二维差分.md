/*
    二维差分
*/
void pre(int n)
{
    vector<vector<int>> diff(n+1,vector<int>(n+1));
    for(int i=1;i<=n;i++)
    {
        int up,down,left,right;
        cin>>up>>down>>left>>right;
        diff[up][left]++;
        diff[down+1][right+1]++;
        diff[up][right+1]--;
        diff[down+1][left]--;
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            diff[i][j]+=diff[i-1][j]+diff[i][j-1]-diff[i-1][j-1];
        }
    }
}
