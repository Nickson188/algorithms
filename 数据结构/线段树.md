const int N=50010;
const int INF=1e9;
int w[N];
struct SGT {
    #define lc u << 1
    #define rc u << 1 | 1
    struct Node {
        int sum;
        int minv;
        int maxv;
        int setv;
        int addv;
    } tr[N * 4];

    void pushup(int u) {
        tr[u].sum = tr[lc].sum + tr[rc].sum;
        tr[u].minv = min(tr[lc].minv, tr[rc].minv);
        tr[u].maxv = max(tr[lc].maxv, tr[rc].maxv);
    }

    void apply_set(int u, int len, int val)
    {
        tr[u].setv = val;
        tr[u].addv = 0;
        tr[u].sum = val * len;
        tr[u].minv = val;
        tr[u].maxv = val;
    }

    void apply_add(int u, int len, int val)
    {
        if (tr[u].setv != -1)
        {
            tr[u].setv += val;
            tr[u].sum = tr[u].setv * len;
            tr[u].minv = tr[u].setv;
            tr[u].maxv = tr[u].setv;
        }
        else
        {
            tr[u].addv += val;
            tr[u].sum += val * len;
            tr[u].minv += val;
            tr[u].maxv += val;
        }
    }

    void pushdown(int u, int l, int r) {
        int mid = (l + r) >> 1;
        int left_len = mid - l + 1;
        int right_len = r - mid;
        if (tr[u].setv != -1)
        {
            apply_set(lc, left_len, tr[u].setv);
            apply_set(rc, right_len, tr[u].setv);
            tr[u].setv = -1;
        }
        if (tr[u].addv != 0)
        {
            apply_add(lc, left_len, tr[u].addv);
            apply_add(rc, right_len, tr[u].addv);
            tr[u].addv = 0;
        }
    }

    void build(int u, int l, int r) {
        tr[u].setv = -1;
        tr[u].addv = 0;
        if (l == r) {
            tr[u].sum = w[l];
            tr[u].minv = w[l];
            tr[u].maxv = w[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(lc, l, mid);
        build(rc, mid + 1, r);
        pushup(u);
    }

    void update_set(int u, int l, int r, int x, int y, int val) {
        if (x > r || y < l) return;
        if (x <= l && r <= y) {
            apply_set(u, r - l + 1, val);
            return;
        }
        pushdown(u, l, r);
        int mid = (l + r) >> 1;
        update_set(lc, l, mid, x, y, val);
        update_set(rc, mid + 1, r, x, y, val);
        pushup(u);
    }

    void update_add(int u, int l, int r, int x, int y, int val) {
        if (x > r || y < l) return;
        if (x <= l && r <= y) {
            apply_add(u, r - l + 1, val);
            return;
        }
        pushdown(u, l, r);
        int mid = (l + r) >> 1;
        update_add(lc, l, mid, x, y, val);
        update_add(rc, mid + 1, r, x, y, val);
        pushup(u);
    }

    int query_min(int u, int l, int r, int x, int y) {
        if (x > r || y < l) return INF;
        if (x <= l && r <= y) return tr[u].minv;
        pushdown(u, l, r);
        int mid = (l + r) >> 1;
        return min(query_min(lc, l, mid, x, y), query_min(rc, mid + 1, r, x, y));
    }

    int query_max(int u, int l, int r, int x, int y) {
        if (x > r || y < l) return -INF;
        if (x <= l && r <= y) return tr[u].maxv;
        pushdown(u, l, r);
        int mid = (l + r) >> 1;
        return max(query_max(lc, l, mid, x, y), query_max(rc, mid + 1, r, x, y));
    }

    int query_sum(int u, int l, int r, int x, int y) {
        if (x > r || y < l) return 0;
        if (x <= l && r <= y) return tr[u].sum;
        pushdown(u, l, r);
        int mid = (l + r) >> 1;
        return query_sum(lc, l, mid, x, y) + query_sum(rc, mid + 1, r, x, y);
    }
} tree;
