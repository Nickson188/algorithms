/*
    单调栈
*/
vector<vector<int>> smaller(int n,vector<int> &a)
{
    vector<int> stk(n+1);
    vector<vector<int>> ans(n+1,vector<int>(2));
    // ans[i][0]存储i左边比a[i]小的最近下标 
    // ans[i][1]存储i右边比a[i]小的最近下标 
    int r=0;
    int cur=0;
    for(int i=1;i<=n;i++)
    {
        // 改为 >= ，维护单调递增栈
        while(r>0 and a[stk[r-1]]>=a[i])
        {
            r--;
            cur=stk[r];
            ans[cur][0]=r>0?stk[r-1]:-1;
            ans[cur][1]=i;
        }
        stk[r++]=i; 
    }
    while(r>0)
    {
        cur=stk[--r];
        ans[cur][0]=r>0?stk[r-1]:-1;
        ans[cur][1]=-1;
    }
    // 处理相等元素：对于相等的元素，右边最近的更小元素应该相同
    for(int i=n-1;i>=1;i--)
    {
        if(ans[i][1]!=-1 and a[ans[i][1]]==a[i])
        {
            ans[i][1]=ans[ans[i][1]][1];
        }
    }
    return ans;
}
