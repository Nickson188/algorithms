typedef long long LL;
const int N=100010,P=1e9+7;
int q,n,m;
LL f[N+5],g[N+5];

LL qsm(LL a,int b){
  LL res=1;
  for(;b;a=a*a%P,b>>=1) if(b&1)res=res*a%P;
  return res;
}
LL C(LL n,LL m){
  if(m < 0 || m > n || n < 0) return 0;
  return f[n]*g[m]%P*g[n-m]%P;
}
void pre()
{
    f[0]=g[0]=1;
    for(int i=1;i<N;i++)
    {
        f[i]=f[i-1]*i%P;
        g[i]=qsm(f[i],P-2)%P;
    }
}
