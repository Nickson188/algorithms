// 空间未优化 MLE
#include<bits/stdc++.h>
using namespace std;

const int N=3410,M=13000;
int n,m;
int v,w,f[N][M];

int main(){
  scanf("%d%d",&n,&m);

  for(int i=1; i<=n; i++){   //枚举物品
    scanf("%d%d",&v,&w);
    for(int j=1; j<=m; j++){ //枚举体积
      if(j<v) f[i][j]=f[i-1][j];            
      else f[i][j]=max(f[i-1][j],f[i-1][j-v]+w);
    }
  } 
  printf("%d\n",f[n][m]);
}

// 滚动数组优化空间
#include<bits/stdc++.h>
using namespace std;

const int N=3410,M=13000;
int n, m;
int v,w,f[2][M];

int main(){
  scanf("%d%d",&n,&m);
    
  for(int i=1; i<=n; i++){   //枚举物品
    scanf("%d%d",&v,&w);
    for(int j=1; j<=m; j++){ //枚举体积
      if(j<v) f[i&1][j]=f[i-1&1][j];            
      else f[i&1][j]=max(f[i-1&1][j],f[i-1&1][j-v]+w); //奇偶滚动
    }
  }
  printf("%d\n",f[n&1][m]);
}
